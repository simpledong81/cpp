在C++中，数学库 <cmath> 提供了许多常用的数学函数，用于执行各种数学运算。以下是常见的数学函数及其用法：

1. 绝对值函数

  • 函数：abs()、fabs()、fabsf()、fabsl()
  • 作用：返回一个数的绝对值。
  • 示例： 
    #include <cmath>
#include <iostream>
using namespace std;

int main() {
    int a = -10;
    double b = -3.14;

    cout << "abs(a): " << abs(a) << endl;  // 输出：10
    cout << "fabs(b): " << fabs(b) << endl;  // 输出：3.14

    return 0;
}

2. 四舍五入函数

  • 函数：round()
  • 作用：返回最接近的整数（四舍五入）。
  • 示例： 
    #include <cmath>
#include <iostream>
using namespace std;

int main() {
    double a = 3.6;
    cout << "round(a): " << round(a) << endl;  // 输出：4

    return 0;
}

3. 下取整函数

  • 函数：floor()
  • 作用：返回不大于给定数的最大整数（向下取整）。
  • 示例： 
    #include <cmath>
#include <iostream>
using namespace std;

int main() {
    double a = 3.6;
    cout << "floor(a): " << floor(a) << endl;  // 输出：3

    return 0;
}

4. 上取整函数

  • 函数：ceil()
  • 作用：返回不小于给定数的最小整数（向上取整）。
  • 示例： 
    #include <cmath>
#include <iostream>
using namespace std;

int main() {
    double a = 3.2;
    cout << "ceil(a): " << ceil(a) << endl;  // 输出：4

    return 0;
}

5. 平方根函数

  • 函数：sqrt()
  • 作用：返回给定数的平方根。
  • 示例： 
    #include <cmath>
#include <iostream>
using namespace std;

int main() {
    double a = 16;
    cout << "sqrt(a): " << sqrt(a) << endl;  // 输出：4

    return 0;
}

6. 常用三角函数

  • 函数：
    • sin()：正弦函数。
    • cos()：余弦函数。
    • tan()：正切函数。
  • 注意：参数以弧度为单位。
  • 示例： 
    #include <cmath>
#include <iostream>
using namespace std;

int main() {
    double angle = 45.0;  // 角度
    double radians = angle * M_PI / 180.0;  // 转换为弧度

    cout << "sin(angle): " << sin(radians) << endl;  // 输出：0.707107
    cout << "cos(angle): " << cos(radians) << endl;  // 输出：0.707107
    cout << "tan(angle): " << tan(radians) << endl;  // 输出：1

    return 0;
}

7. 对数函数

  • 函数：
    • log()：自然对数（以 e 为底）。
    • log10()：常用对数（以 10 为底）。
  • 示例： 
    #include <cmath>
#include <iostream>
using namespace std;

int main() {
    double a = 2.71828;  // e 的近似值

    cout << "log(a): " << log(a) << endl;  // 输出：1（ln(e) = 1）
    cout << "log10(100): " << log10(100) << endl;  // 输出：2

    return 0;
}

8. 指数函数

  • 函数：exp()
  • 作用：返回 e 的给定次幂。
  • 示例： 
    #include <cmath>
#include <iostream>
using namespace std;

int main() {
    double a = 1.0;
    cout << "exp(a): " << exp(a) << endl;  // 输出：2.71828（e 的近似值）

    return 0;
}

9. 综合示例

以下是一个完整的C++程序，演示了各种数学函数的用法：

#include <cmath>
#include <iostream>
using namespace std;

int main() {
    double a = -3.14;
    double b = 2.71828;
    double angle = 45.0;  // 角度

    // 绝对值
    cout << "abs(a): " << abs(a) << endl;  // 输出：3.14

    // 四舍五入
    cout << "round(a): " << round(a) << endl;  // 输出：-3

    // 下取整
    cout << "floor(a): " << floor(a) << endl;  // 输出：-4

    // 上取整
    cout << "ceil(a): " << ceil(a) << endl;  // 输出：-3

    // 平方根
    cout << "sqrt(b): " << sqrt(b) << endl;  // 输出：1.64872

    // 三角函数
    double radians = angle * M_PI / 180.0;  // 转换为弧度
    cout << "sin(angle): " << sin(radians) << endl;  // 输出：0.707107
    cout << "cos(angle): " << cos(radians) << endl;  // 输出：0.707107
    cout << "tan(angle): " << tan(radians) << endl;  // 输出：1

    // 对数函数
    cout << "log(b): " << log(b) << endl;  // 输出：1（ln(e) = 1）
    cout << "log10(100): " << log10(100) << endl;  // 输出：2

    // 指数函数
    cout << "exp(1): " << exp(1) << endl;  // 输出：2.71828

    return 0;
}
