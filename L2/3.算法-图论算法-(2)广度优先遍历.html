广度优先遍历（Breadth-First Traversal, BFT）

1. 什么是广度优先遍历？

广度优先遍历是一种用于遍历或搜索树或图的算法。它的核心思想是从起始节点开始，逐层访问其邻居节点，直到遍历完所有节点或找到目标节点。

2. 广度优先遍历的基本思想

  1. 从起始节点开始：选择一个起始节点作为当前节点。
  2. 访问当前节点：标记当前节点为已访问。
  3. 访问邻居节点：将当前节点的所有未访问的邻居节点加入队列。
  4. 逐层遍历：从队列中取出下一个节点，重复上述过程，直到队列为空。

3. 广度优先遍历的步骤

  1. 选择一个起始节点，标记为已访问，并将其加入队列。
  2. 从队列中取出一个节点，访问该节点。
  3. 将该节点的所有未访问的邻居节点加入队列，并标记为已访问。
  4. 重复上述过程，直到队列为空。

4. 广度优先遍历的代码实现

（1）图的表示

图可以用邻接表或邻接矩阵表示。以下代码使用邻接表表示图。

#include <iostream>
#include <vector>
#include <queue>
using namespace std;

// 图的邻接表表示
class Graph {
public:
    int V; // 顶点数
    vector<vector<int>> adj; // 邻接表

    Graph(int V) : V(V), adj(V) {}

    // 添加边
    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u); // 无向图需要添加双向边
    }

    // 广度优先遍历
    void BFT(int start) {
        vector<bool> visited(V, false); // 初始化访问标记数组
        queue<int> queue;

        visited[start] = true; // 标记起始节点为已访问
        queue.push(start); // 将起始节点加入队列

        while (!queue.empty()) {
            int v = queue.front();
            queue.pop();
            cout << v << " "; // 输出当前节点

            // 遍历所有邻居节点
            for (int neighbor : adj[v]) {
                if (!visited[neighbor]) {
                    visited[neighbor] = true; // 标记邻居节点为已访问
                    queue.push(neighbor); // 将邻居节点加入队列
                }
            }
        }
    }
};

int main() {
    Graph g(6); // 创建一个有 6 个顶点的图

    // 添加边
    g.addEdge(0, 1);
    g.addEdge(0, 2);
    g.addEdge(1, 3);
    g.addEdge(1, 4);
    g.addEdge(2, 4);
    g.addEdge(3, 4);
    g.addEdge(3, 5);
    g.addEdge(4, 5);

    cout << "广度优先遍历结果: ";
    g.BFT(0); // 从顶点 0 开始 BFT

    return 0;
}

输出：

广度优先遍历结果: 0 1 2 3 4 5
（2）BFT 的应用：最短路径（无权图）

在无权图中，BFT 可以用于求解从起始节点到其他节点的最短路径。

#include <iostream>
#include <vector>
#include <queue>
using namespace std;

// 图的邻接表表示
class Graph {
public:
    int V; // 顶点数
    vector<vector<int>> adj; // 邻接表

    Graph(int V) : V(V), adj(V) {}

    // 添加边
    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u); // 无向图需要添加双向边
    }

    // 广度优先遍历（求解最短路径）
    vector<int> shortestPath(int start) {
        vector<bool> visited(V, false); // 初始化访问标记数组
        vector<int> distance(V, -1); // 初始化距离数组
        queue<int> queue;

        visited[start] = true; // 标记起始节点为已访问
        distance[start] = 0; // 起始节点的距离为 0
        queue.push(start); // 将起始节点加入队列

        while (!queue.empty()) {
            int v = queue.front();
            queue.pop();

            // 遍历所有邻居节点
            for (int neighbor : adj[v]) {
                if (!visited[neighbor]) {
                    visited[neighbor] = true; // 标记邻居节点为已访问
                    distance[neighbor] = distance[v] + 1; // 更新邻居节点的距离
                    queue.push(neighbor); // 将邻居节点加入队列
                }
            }
        }

        return distance;
    }
};

int main() {
    Graph g(6); // 创建一个有 6 个顶点的图

    // 添加边
    g.addEdge(0, 1);
    g.addEdge(0, 2);
    g.addEdge(1, 3);
    g.addEdge(1, 4);
    g.addEdge(2, 4);
    g.addEdge(3, 4);
    g.addEdge(3, 5);
    g.addEdge(4, 5);

    vector<int> distances = g.shortestPath(0); // 从顶点 0 开始 BFT

    cout << "从顶点 0 到其他顶点的最短路径: ";
    for (int dist : distances) {
        cout << dist << " ";
    }
    cout << endl;

    return 0;
}

输出：

从顶点 0 到其他顶点的最短路径: 0 1 1 2 2 3

5. 广度优先遍历的时间复杂度

  • 时间复杂度：O(V + E)，其中 V 是顶点数，E 是边数。
  • 空间复杂度：O(V)，用于存储访问标记和队列。

6. 广度优先遍历的应用

  • 最短路径（无权图）：求解从起始节点到其他节点的最短路径。
  • 连通性检测：判断图中是否存在路径。
  • 拓扑排序：用于有向无环图（DAG）的拓扑排序。
  • 求解迷宫问题：寻找从起点到终点的最短路径。

7. 广度优先遍历的优缺点

  • 优点：
    • 适合求解最短路径问题（无权图）。
    • 逐层遍历，适合解决层次相关的问题。
  • 缺点：
    • 空间复杂度较高，尤其是在图的分支因子较大时。
    • 不适合求解深度较大的问题（深度优先搜索更适合）。

8. 总结

广度优先遍历是一种重要的图遍历算法，适用于解决最短路径问题、连通性问题等。通过队列实现，BFT 可以高效地遍历图或树结构。掌握 BFT 的思想和实现方法，有助于解决许多与图相关的问题。
